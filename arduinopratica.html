        t num = 0;

void setup()
{
  pinMode(12, OUTPUT);
  pinMode(10, OUTPUT);
  pinMode(2, OUTPUT);
  pinMode(6, OUTPUT);
  pinMode(4, OUTPUT);
  pinMode(8, OUTPUT);

  num = 0;
}

void loop()
{
  while (num < 2) {
    digitalWrite(12, HIGH);
    digitalWrite(10, LOW);
    digitalWrite(2, HIGH);
    digitalWrite(6, LOW);
    delay(5000); // Wait for 5000 millisecond(s)
    digitalWrite(2, LOW);
    digitalWrite(4, HIGH);
    delay(2000); // Wait for 2000 millisecond(s)
    digitalWrite(4, LOW);
    digitalWrite(6, HIGH);
    digitalWrite(12, LOW);
    digitalWrite(8, HIGH);
    delay(5000); // Wait for 5000 millisecond(s)
    digitalWrite(8, LOW);
    digitalWrite(10, HIGH);
    delay(2000); // Wait for 2000 millisecond(s)
    num += 1;
  }
  if (num == 2) {
    digitalWrite(12, LOW);
    digitalWrite(6, LOW);
    digitalWrite(2, LOW);
    digitalWrite(4, HIGH);
    digitalWrite(10, HIGH);
    delay(1000); // Wait for 1000 millisecond(s)
    digitalWrite(10, LOW);
    digitalWrite(4, LOW);
    delay(1000); // Wait for 1000 millisecond(s)
  }
}
